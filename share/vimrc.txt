" 获取绝对路径的函数
function! LightlineFilepath()
    return expand('%:p')
endfunction

" 获取 Git 状态的函数
function! LightlineGitStatus()
    if exists('*FugitiveHead')
        " 获取当前分支名称
        let l:branch = FugitiveHead()
        " 获取当前文件绝对路径
        let l:file = expand('%:p')
        " 如果文件不存在或不可读
        if empty(l:file) || !filereadable(l:file)
            return (empty(l:branch) ? '(no branch)' : l:branch) . ':Missing'
        endif
        " 查询当前文件的 Git 状态
        let l:status = system('git status --porcelain ' . shellescape(l:file))
        let l:status = substitute(l:status, '\n', '', 'g')
        let l:state = ''
        if l:status =~ '^??'
            let l:state = 'Untracked'
        elseif l:status =~ '^!!'
            let l:state = 'Ignored'
        elseif l:status =~ '^M ' || l:status =~ '^ M'
            let l:state = 'Modified'
        elseif l:status =~ '^A '
            let l:state = 'Staged'
        elseif l:status =~ '^D '
            let l:state = 'Deleted'
        elseif l:status =~ '^R '
            let l:state = 'Renamed'
        elseif l:status =~ '^C '
            let l:state = 'Copied'
        elseif l:status =~ '^UU'
            let l:state = 'Unmerged'
        else
            let l:state = 'Clean'
        endif
        return (empty(l:branch) ? '(no branch)' : l:branch) . ':' . l:state
    endif
    return ''
endfunction
